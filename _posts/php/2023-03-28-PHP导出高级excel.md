2  php 写法
<pre><code>    
  /**
 * 导出文件
 * @return string
 */
public  function  export()
{
$file_name = "成绩单-".date("Y-m-d H:i:s",time());
$file_suffix = "xls";
$data['a']=1;
$data['b']="zhangsan";
header("Content-Type: application/vnd.ms-excel");
header("Content-Disposition: attachment; filename=$file_name.$file_suffix");
//根据业务，自己进行模板赋值。
$this->display("1.php",$data);
}
